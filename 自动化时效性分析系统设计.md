# 自动化时效性分析系统设计

## 一、系统架构

```
输入数据 (JSON/CSV)
    ↓
数据加载与预处理模块
    ↓
特征计算模块
    ↓
时效性评估模块 (打分)
    ↓
原因诊断模块 (三层诊断)
    ↓
报告生成模块
    ↓
输出结果 (JSON + HTML报告)
```

### 4.1 数据预处理模块

```python
class DataPreprocessor:
    """数据预处理模块"""
    
    def __init__(self, config):
        self.config = config
    
    def load_data(self, file_path):
        """加载数据"""
        if file_path.endswith('.json'):
            return self._load_json(file_path)
        elif file_path.endswith('.csv'):
            return self._load_csv(file_path)
        else:
            raise ValueError("Unsupported file format")
    
    def preprocess(self, raw_data):
        """预处理：计算doc年龄等基础字段"""
        processed_data = []
        
        for item in raw_data:
            processed_item = self._process_single_query(item)
            processed_data.append(processed_item)
        
        return processed_data
    
    def _process_single_query(self, item):
        """处理单个query"""
        search_time = pd.to_datetime(item['search_time'])
        
        # 处理engine1
        item['engine1']['docs'] = self._add_doc_age(
            item['engine1']['docs'], search_time
        )
        
        # 处理engine2
        item['engine2']['docs'] = self._add_doc_age(
            item['engine2']['docs'], search_time
        )
        
        # 处理test
        item['test']['docs'] = self._add_doc_age(
            item['test']['docs'], search_time
        )
        
        return item
    
    def _add_doc_age(self, docs, search_time):
        """添加doc年龄字段"""
        for doc in docs:
            publish_time = pd.to_datetime(doc['publish_time'])
            doc['age_days'] = (search_time - publish_time).total_seconds() / 86400
            doc['age_hours'] = (search_time - publish_time).total_seconds() / 3600
        return docs
```

### 4.2 特征计算模块

```python
class FeatureExtractor:
    """特征提取模块"""
    
    def __init__(self, config):
        self.config = config
        self.top_k_values = config['features']['top_k_values']
        self.time_windows = config['features']['time_windows']
    
    def extract_features(self, query_data):
        """提取单个query的所有特征"""
        features = {
            'query': query_data['query'],
            'search_time': query_data['search_time']
        }
        
        # 提取三个引擎的特征
        for engine_name in ['engine1', 'engine2', 'test']:
            engine_data = query_data[engine_name]
            docs = engine_data['docs']
            
            prefix = f"{engine_name}_"
            
            # 基础统计特征
            features[f"{prefix}total_docs"] = len(docs)
            
            # 不同TOP-K的特征
            for k in self.top_k_values:
                top_k_docs = docs[:k]
                features.update(
                    self._calc_topk_features(top_k_docs, f"{prefix}top{k}")
                )
            
            # 时效卡特征（仅engine1和engine2）
            if engine_name in ['engine1', 'engine2']:
                features[f"{prefix}has_card"] = engine_data['has_timeliness_card']
        
        # 计算baseline特征（engine1和engine2的平均）
        features.update(self._calc_baseline_features(features))
        
        # 计算test特有特征
        features['test_query_level'] = query_data['test']['query_timeliness_level']
        features.update(self._calc_test_specific_features(query_data['test']['docs']))
        
        return features
    
    def _calc_topk_features(self, docs, prefix):
        """计算TOP-K的统计特征"""
        if not docs:
            return {}
        
        ages = [doc['age_days'] for doc in docs]
        
        features = {
            f"{prefix}_avg_age": np.mean(ages),
            f"{prefix}_median_age": np.median(ages),
            f"{prefix}_min_age": np.min(ages),
            f"{prefix}_max_age": np.max(ages),
            f"{prefix}_std_age": np.std(ages) if len(ages) > 1 else 0,
        }
        
        # 不同时间窗口的新鲜文档占比
        for window in self.time_windows:
            fresh_count = sum(1 for age in ages if age <= window)
            features[f"{prefix}_fresh_{window}d_ratio"] = fresh_count / len(ages)
        
        return features
    
    def _calc_baseline_features(self, features):
        """计算baseline特征（engine1和engine2的平均）"""
        baseline = {}
        
        for k in self.top_k_values:
            # 平均doc年龄
            e1_avg = features[f"engine1_top{k}_avg_age"]
            e2_avg = features[f"engine2_top{k}_avg_age"]
            baseline[f"baseline_top{k}_avg_age"] = (e1_avg + e2_avg) / 2
            
            # 最小doc年龄
            e1_min = features[f"engine1_top{k}_min_age"]
            e2_min = features[f"engine2_top{k}_min_age"]
            baseline[f"baseline_top{k}_min_age"] = min(e1_min, e2_min)
            
            # 新鲜文档占比
            for window in self.time_windows:
                e1_ratio = features[f"engine1_top{k}_fresh_{window}d_ratio"]
                e2_ratio = features[f"engine2_top{k}_fresh_{window}d_ratio"]
                baseline[f"baseline_top{k}_fresh_{window}d_ratio"] = (e1_ratio + e2_ratio) / 2
        
        # 时效卡标签
        baseline['baseline_has_timeliness_need'] = (
            features['engine1_has_card'] or features['engine2_has_card']
        )
        
        return baseline
    
    def _calc_test_specific_features(self, test_docs):
        """计算test引擎特有的特征（与时效性打分相关）"""
        if not test_docs:
            return {}
        
        ages = [doc['age_days'] for doc in test_docs]
        scores = [doc['timeliness_score'] for doc in test_docs]
        
        # 计算相关系数
        correlation = np.corrcoef(ages, scores)[0, 1] if len(ages) > 1 else 0
        
        # 新旧文档打分对比
        fresh_docs = [doc for doc in test_docs if doc['age_days'] <= 7]
        old_docs = [doc for doc in test_docs if doc['age_days'] > 30]
        
        fresh_avg_score = np.mean([d['timeliness_score'] for d in fresh_docs]) if fresh_docs else 0
        old_avg_score = np.mean([d['timeliness_score'] for d in old_docs]) if old_docs else 0
        
        return {
            'test_score_age_correlation': correlation,
            'test_score_std': np.std(scores),
            'test_fresh_avg_score': fresh_avg_score,
            'test_old_avg_score': old_avg_score,
            'test_fresh_old_gap': fresh_avg_score - old_avg_score,
            'test_fresh_doc_count': len(fresh_docs),
            'test_old_doc_count': len(old_docs)
        }
```

### 4.3 时效性评估模块

```python
class TimelinessEvaluator:
    """时效性评估模块"""
    
    def __init__(self, config):
        self.config = config
        self.weights = config['thresholds']['weights']
        self.poor_threshold = config['thresholds']['poor_score']
        self.fair_threshold = config['thresholds']['fair_score']
    
    def evaluate(self, features):
        """评估时效性得分"""
        scores = {}
        
        # 维度1：平均doc年龄得分
        scores['dimension1'] = self._calc_avg_age_score(features)
        
        # 维度2：新鲜文档覆盖率得分
        scores['dimension2'] = self._calc_fresh_ratio_score(features)
        
        # 维度3：最新文档发现能力得分
        scores['dimension3'] = self._calc_min_age_score(features)
        
        # 维度4：时效卡得分
        scores['dimension4'] = self._calc_card_score(features)
        
        # 综合得分
        total_score = (
            scores['dimension1'] * self.weights['avg_age'] +
            scores['dimension2'] * self.weights['fresh_ratio'] +
            scores['dimension3'] * self.weights['min_age'] +
            scores['dimension4'] * self.weights['card']
        )
        
        # 判定标签
        if total_score >= self.fair_threshold:
            label = "时效性良好"
        elif total_score >= self.poor_threshold:
            label = "时效性一般"
        else:
            label = "时效性差"
        
        return {
            'total_score': total_score,
            'label': label,
            'dimension_scores': scores
        }
    
    def _calc_avg_age_score(self, features):
        """计算平均doc年龄得分（使用TOP5）"""
        test_avg = features['test_top5_avg_age']
        baseline_avg = features['baseline_top5_avg_age']
        
        if baseline_avg == 0:
            return 100
        
        age_ratio = test_avg / baseline_avg
        
        if age_ratio <= 1.0:
            return 100
        elif age_ratio <= 1.5:
            return 100 - (age_ratio - 1.0) * 100
        elif age_ratio <= 2.0:
            return 50 - (age_ratio - 1.5) * 60
        else:
            return max(0, 20 - (age_ratio - 2.0) * 20)
    
    def _calc_fresh_ratio_score(self, features):
        """计算新鲜文档覆盖率得分（使用7天窗口）"""
        test_ratio = features['test_top5_fresh_7d_ratio']
        baseline_ratio = features['baseline_top5_fresh_7d_ratio']
        
        fresh_diff = test_ratio - baseline_ratio
        
        if fresh_diff >= 0:
            return 100
        elif fresh_diff >= -0.2:
            return 100 + fresh_diff * 250
        elif fresh_diff >= -0.4:
            return 50 + (fresh_diff + 0.2) * 150
        else:
            return max(0, 50 + (fresh_diff + 0.4) * 100)
    
    def _calc_min_age_score(self, features):
        """计算最新文档发现能力得分"""
        test_min = features['test_top5_min_age']
        baseline_min = features['baseline_top5_min_age']
        
        if baseline_min == 0:
            return 100
        
        min_age_ratio = test_min / baseline_min
        
        if min_age_ratio <= 1.2:
            return 100
        elif min_age_ratio <= 2.0:
            return 100 - (min_age_ratio - 1.2) * 62.5
        elif min_age_ratio <= 5.0:
            return 50 - (min_age_ratio - 2.0) * 13.3
        else:
            return 0
    
    def _calc_card_score(self, features):
        """计算时效卡得分"""
        if not features['baseline_has_timeliness_need']:
            return 100
        
        # 如果baseline需要时效卡，检查test的综合表现
        # 这里简化为：如果前三个维度都表现不错，认为有时效卡效果
        # 实际可根据test的其他信号判断
        return 100  # 简化处理
```

### 4.4 原因诊断模块

```python
class ReasonDiagnoser:
    """原因诊断模块"""
    
    def __init__(self, config):
        self.config = config
        self.thresholds = config['thresholds']
    
    def diagnose(self, query_data, features, evaluation):
        """诊断时效性差的原因"""
        # 只诊断时效性差的query
        if evaluation['label'] != "时效性差":
            return None
        
        # 按优先级诊断
        # 原因1：识别错误（最高优先级）
        reason1 = self._diagnose_recognition_error(query_data, features)
        if reason1['flag']:
            return reason1
        
        # 原因2：打分公式（第二优先级）
        reason2 = self._diagnose_scoring_formula(query_data, features)
        if reason2['flag']:
            return reason2
        
        # 原因3：融合策略（第三优先级）
        reason3 = self._diagnose_fusion_strategy(query_data, features)
        if reason3['flag']:
            return reason3
        
        # 如果都不是，标记为未知原因
        return {
            'flag': True,
            'reason': '原因未知',
            'details': {}
        }
    
    def _diagnose_recognition_error(self, query_data, features):
        """诊断原因1：query时效性等级识别错误"""
        cfg = self.thresholds['recognition']
        
        # 判断是否应该是高时效
        should_be_high = False
        confidence = "低"
        evidence = []
        
        # 证据1：时效卡
        has_both_cards = features['engine1_has_card'] and features['engine2_has_card']
        has_one_card = features['engine1_has_card'] or features['engine2_has_card']
        
        if has_both_cards:
            should_be_high = True
            confidence = "高"
            evidence.append("两个权威引擎都有时效卡")
        elif has_one_card:
            confidence = "中"
            evidence.append("一个权威引擎有时效卡")
        
        # 证据2：平均doc年龄
        baseline_avg = features['baseline_top5_avg_age']
        
        if baseline_avg <= cfg['fresh_age_strong']:
            freshness_pref = "强"
            evidence.append(f"权威引擎平均doc年龄≤{cfg['fresh_age_strong']}天")
            if confidence == "中":
                should_be_high = True
        elif baseline_avg <= cfg['fresh_age_medium']:
            freshness_pref = "中"
            evidence.append(f"权威引擎平均doc年龄≤{cfg['fresh_age_medium']}天")
        else:
            freshness_pref = "弱"
        
        # 证据3：新鲜文档占比
        baseline_fresh_ratio = features['baseline_top5_fresh_7d_ratio']
        
        if baseline_fresh_ratio >= cfg['fresh_ratio_high']:
            fresh_concentration = "高"
            evidence.append(f"权威引擎新鲜文档占比≥{cfg['fresh_ratio_high']}")
            if freshness_pref == "强":
                should_be_high = True
        elif baseline_fresh_ratio >= cfg['fresh_ratio_medium']:
            fresh_concentration = "中"
        else:
            fresh_concentration = "低"
        
        # 判定
        test_level = features['test_query_level']
        
        if should_be_high and test_level == "其他时效":
            return {
                'flag': True,
                'reason': '原因1_识别错误',
                'sub_reason': '高时效query被误判为其他时效',
                'details': {
                    '时效性需求置信度': confidence,
                    '证据': evidence,
                    'baseline平均doc年龄': baseline_avg,
                    'baseline新鲜文档占比': baseline_fresh_ratio,
                    'engine1有时效卡': features['engine1_has_card'],
                    'engine2有时效卡': features['engine2_has_card'],
                    'test判别结果': test_level,
                    '正确判别': '高时效'
                }
            }
        
        return {'flag': False}
    
    def _diagnose_scoring_formula(self, query_data, features):
        """诊断原因2：doc时效性打分公式不合理"""
        test_level = features['test_query_level']
        
        # 根据query类型选择阈值
        if test_level == "高时效":
            cfg = self.thresholds['scoring']['high_timeliness']
        else:
            cfg = self.thresholds['scoring']['other_timeliness']
        
        problems = []
        
        # 检查1：相关系数
        corr = features['test_score_age_correlation']
        if corr < cfg['corr_min'] or corr > cfg['corr_max']:
            problems.append({
                '问题': 'doc年龄与时效性打分相关性异常',
                '实际相关系数': round(corr, 3),
                '期望范围': f"[{cfg['corr_min']}, {cfg['corr_max']}]",
                '严重程度': 'high'
            })
        
        # 检查2：打分区分度
        score_std = features['test_score_std']
        if score_std < cfg['std_min']:
            problems.append({
                '问题': '时效性打分区分度不足',
                '实际标准差': round(score_std, 3),
                '最小期望': cfg['std_min'],
                '严重程度': 'medium'
            })
        
        # 检查3：新旧文档打分差距
        gap = features['test_fresh_old_gap']
        if features['test_fresh_doc_count'] > 0 and features['test_old_doc_count'] > 0:
            if gap < cfg['gap_min']:
                problems.append({
                    '问题': '新旧文档打分差距不足',
                    '实际差距': round(gap, 3),
                    '最小期望': cfg['gap_min'],
                    '新文档平均分': round(features['test_fresh_avg_score'], 3),
                    '旧文档平均分': round(features['test_old_avg_score'], 3),
                    '严重程度': 'high'
                })
        
        # 检查4：新文档排序位置（通过对比baseline）
        test_fresh_ratio = features['test_top5_fresh_7d_ratio']
        baseline_fresh_ratio = features['baseline_top5_fresh_7d_ratio']
        
        if test_fresh_ratio < baseline_fresh_ratio * (1 / self.thresholds['scoring']['rank_ratio_max']):
            problems.append({
                '问题': '新文档打分导致排序靠后',
                'test新文档占比': round(test_fresh_ratio, 3),
                'baseline新文档占比': round(baseline_fresh_ratio, 3),
                '差距比例': round(baseline_fresh_ratio / test_fresh_ratio if test_fresh_ratio > 0 else 999, 2),
                '严重程度': 'high'
            })
        
        if problems:
            return {
                'flag': True,
                'reason': '原因2_打分公式',
                'sub_reason': '时效性打分计算公式不合理',
                'details': {
                    'query时效性等级': test_level,
                    '问题列表': problems,
                    '问题数量': len(problems)
                }
            }
        
        return {'flag': False}
    
    def _diagnose_fusion_strategy(self, query_data, features):
        """诊断原因3：融合策略不合理"""
        cfg = self.thresholds['fusion']
        problems = []
        
        # 检查1：计算理想排序的提升潜力
        test_docs = query_data['test']['docs']
        
        # 按时效性打分排序（理想排序）
        sorted_docs = sorted(test_docs, key=lambda x: x['timeliness_score'], reverse=True)
        ideal_top5_avg_age = np.mean([doc['age_days'] for doc in sorted_docs[:5]])
        
        # 当前排序（假设按返回顺序）
        actual_top5_avg_age = features['test_top5_avg_age']
        
        if actual_top5_avg_age > 0:
            improvement_potential = (actual_top5_avg_age - ideal_top5_avg_age) / actual_top5_avg_age
            
            if improvement_potential > cfg['improvement_potential']:
                problems.append({
                    '问题': '时效性打分在融合中权重过低',
                    '当前平均doc年龄': round(actual_top5_avg_age, 2),
                    '理想平均doc年龄': round(ideal_top5_avg_age, 2),
                    '提升潜力': f"{improvement_potential:.1%}",
                    '严重程度': 'high'
                })
        
        # 检查2：新文档是否被压制
        fresh_docs = [doc for doc in test_docs if doc['age_days'] <= 7]
        if fresh_docs:
            # 新文档的平均打分
            fresh_avg_score = np.mean([doc['timeliness_score'] for doc in fresh_docs])
            all_avg_score = np.mean([doc['timeliness_score'] for doc in test_docs])
            
            # 新文档在top10中的占比
            fresh_in_top10 = sum(1 for doc in test_docs[:10] if doc['age_days'] <= 7)
            fresh_ratio_in_top10 = fresh_in_top10 / min(10, len(test_docs))
            
            expected_ratio = len(fresh_docs) / len(test_docs)
            
            if fresh_avg_score > all_avg_score + 0.1 and fresh_ratio_in_top10 < expected_ratio * cfg['fresh_suppression_ratio']:
                problems.append({
                    '问题': '新文档召回但被融合策略压制',
                    '新文档平均时效分': round(fresh_avg_score, 3),
                    '全部文档平均分': round(all_avg_score, 3),
                    '新文档在top10占比': round(fresh_ratio_in_top10, 3),
                    '期望占比': round(expected_ratio, 3),
                    '严重程度': 'medium'
                })
        
        if problems:
            return {
                'flag': True,
                'reason': '原因3_融合策略',
                'sub_reason': '时效性打分融合权重不合理',
                'details': {
                    '问题列表': problems,
                    '问题数量': len(problems)
                }
            }
        
        return {'flag': False}
```

## 五、主程序流程

```python
class TimelinessAnalysisSystem:
    """时效性分析系统主类"""
    
    def __init__(self, config_path='config.yaml'):
        # 加载配置
        with open(config_path, 'r', encoding='utf-8') as f:
            self.config = yaml.safe_load(f)
        
        # 初始化各模块
        self.preprocessor = DataPreprocessor(self.config)
        self.feature_extractor = FeatureExtractor(self.config)
        self.evaluator = TimelinessEvaluator(self.config)
        self.diagnoser = ReasonDiagnoser(self.config)
        self.reporter = ReportGenerator(self.config)
    
    def run(self, input_file, output_dir='./output'):
        """运行完整分析流程"""
        print("=" * 50)
        print("搜索引擎时效性分析系统")
        print("=" * 50)
        
        # Step 1: 数据加载与预处理
        print("\n[1/5] 数据加载与预处理...")
        raw_data = self.preprocessor.load_data(input_file)
        processed_data = self.preprocessor.preprocess(raw_data)
        print(f"✓ 成功加载 {len(processed_data)} 个query")
        
        # Step 2: 特征提取
        print("\n[2/5] 特征提取...")
        all_results = []
        
        for idx, query_data in enumerate(processed_data, 1):
            if idx % 100 == 0:
                print(f"  处理进度: {idx}/{len(processed_data)}")
            
            # 提取特征
            features = self.feature_extractor.extract_features(query_data)
            
            # 评估时效性
            evaluation = self.evaluator.evaluate(features)
            
            # 诊断原因（仅对时效性差的query）
            diagnosis = None
            if evaluation['label'] == '时效性差':
                diagnosis = self.diagnoser.diagnose(query_data, features, evaluation)
            
            # 汇总结果
            result = {
                'query': query_data['query'],
                'features': features,
                'evaluation': evaluation,
                'diagnosis': diagnosis
            }
            
            all_results.append(result)
        
        print(f"✓ 特征提取完成")
        
        # Step 3: 生成报告
        print("\n[3/5] 生成分析报告...")
        report = self.reporter.generate(all_results)
        print(f"✓ 发现 {report['summary']['poor_timeliness_queries']} 个时效性差的query")
        
        # Step 4: 导出结果
        print("\n[4/5] 导出结果...")
        os.makedirs(output_dir, exist_ok=True)
        
        if self.config['output']['export_json']:
            json_path = os.path.join(output_dir, 'analysis_report.json')
            self.reporter.export_json(report, json_path)
            print(f"✓ JSON报告: {json_path}")
        
        if self.config['output']['export_html']:
            html_path = os.path.join(output_dir, 'analysis_report.html')
            self.reporter.export_html(report, html_path)
            print(f"✓ HTML报告: {html_path}")
        
        if self.config['output']['export_csv']:
            csv_path = os.path.join(output_dir, 'poor_queries.csv')
            self.reporter.export_csv(report, csv_path)
            print(f"✓ CSV报告: {csv_path}")
        
        # Step 5: 打印摘要
        print("\n[5/5] 分析摘要")
        print("=" * 50)
        self._print_summary(report['summary'])
        
        return report
    
    def _print_summary(self, summary):
        """打印摘要信息"""
        print(f"总query数: {summary['total_queries']}")
        print(f"时效性差: {summary['poor_timeliness_queries']} ({summary['poor_ratio']:.1%})")
        print("\n原因分布:")
        for reason, stats in summary['reason_distribution'].items():
            print(f"  {reason}: {stats['count']} ({stats['ratio']:.1%})")


# 主入口
```
